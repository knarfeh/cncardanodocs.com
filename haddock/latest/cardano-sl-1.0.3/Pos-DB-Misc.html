getProxySecretKeysLight :: MonadDB m => m [ProxySKLight] Source #

Gets proxy secret keys stored by node. We store only "related" psks: those who have pskIssuer or pskDelegate matching our pk.

addProxySecretKey :: MonadDB m => ProxySKLight -> m () Source #

Adds proxy secret key if not present. Nothing if present. Call this under write lock only (see miscLock).

removeProxySecretKey :: MonadDB m => PublicKey -> m () Source #

Removes proxy secret key if present by issuer pk. Call it under write lock only (see miscLock).

dropOldProxySecretKeys :: MonadDB m => EpochIndex -> m () Source #

Given epochindex, throws away all outdated PSKs. Remark: it doesn't remove keys that can be used in future. Call it under write lock only (see miscLock).
